Sub FontToggler()
	Select Case Selection.Font.Name
		Case "SolaimanLipi"
				Selection.Font.Name = "Cambria Math"
		Case "Cambria Math"
			Selection.Font.Name = "SolaimanLipi"
		Case Else
			Selection.Font.Name = "Cambria Math"
	End Select
End Sub
Sub ColorToggler()
	Select Case Selection.Font.Color
		Case wdColorRed
			Selection.Font.Color = wdColorAuto
		Case wdColorAuto
			Selection.Font.Color = wdColorRed
		Case Else
			Selection.Font.Color = wdColorAuto
	End Select
End Sub
